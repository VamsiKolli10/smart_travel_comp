function deg2rad(d) {
  return d * (Math.PI / 180);
}
function kmBetween(a, b) {
  const R = 6371;
  const dLat = deg2rad(b.lat - a.lat);
  const dLng = deg2rad(b.lng - a.lng);
  const s1 =
    Math.sin(dLat / 2) ** 2 +
    Math.cos(deg2rad(a.lat)) *
      Math.cos(deg2rad(b.lat)) *
      Math.sin(dLng / 2) ** 2;
  return 2 * R * Math.atan2(Math.sqrt(s1), Math.sqrt(1 - s1));
}

function parseCityStateCountry(address, fallbackCity = "") {
  if (!address || typeof address !== "string") {
    const label = (fallbackCity || "").trim();
    return { city: label, state: "", country: "" };
  }
 
  const parts = address
    .split(",")
    .map((segment) => segment.trim())
    .filter(Boolean);
 
  if (!parts.length) {
    const label = (fallbackCity || "").trim();
    return { city: label, state: "", country: "" };
  }
 
  const country = parts.pop() || "";
  let state = "";
  if (parts.length >= 2) {
    state = parts.pop() || "";
  }
 
  let city = parts.join(", ");
  if (!city) {
    city = parts[0] || "";
  }
  if (!city && fallbackCity) {
    city = fallbackCity.trim();
  }
 
  return { city: city.trim(), state: state.trim(), country: country.trim() };
}
